
// Endpoint models for REST interface:

// Return all of the available cities.
// Endpoint: /cities

[
	{
		"id": <Integer>,
		"city_name": <String>
	},

	{
		"id": <Integer>,
		"city_name": <String>
	}
]


// Return possible vehicle types for the given city.
// Endpoint: /cities/{:id}

{
	"id": <Integer>,
	"city_name": <String>,
	"vehicles": [
		{
			"id": <Integer>,
			"vehicle_type": <String>,
		},
		..
	]
}


// Return the fees associated with all vehicles associated with this city.
// Endpoint: /cities/{:id}/vehicles

[
	{
		"id": <Integer>,
		"vehicle_type": <String>,
		"fees": {
				"regional_base_fee": <BigDecimal>,
				"air_temp_extra_fees": {
					"minus_10_or_lower": <BigDecimal>,
					"between_minus_10_and_0": <BigDecimal>,
			},
			
			// NB: Wind speed fee exists only for vehicle_type "Bike"
			// Above 20 m/s the selected vehicle type should be forbidden.
			"wind_speed_extra_fees": {
				"between_10_and_20": <BigDecimal>,
			},
			
			// NB: Weather phenomenon fee exists only for vehicle_type "Bike" or "Scooter"
			// with glaze, hail or thunder the selected vehicle types should be forbidden.
			"weather_phenomenon_extra_fees": {
				"with_snow": <BigDecimal>,
				"with_sleet": <BigDecimal>,
				"with_rain": <BigDecimal>,
			}
		}
	}
	..
]


// Return the calculated delivery fee for the given vehicle type.
// Endpoint: /cities/{:id}/vehicles/{:id}

{
	"id": <Integer>,
	"city_name": <String>,
	"vehicle_type": <String>,
	"total_delivery_fee": <BigDecimal>
}
